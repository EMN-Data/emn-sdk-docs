[docs]    def get_projects(
        self, permission: str = "create_dataset", fields: Sequence[str] = ()
    ) -> Dict[str, str]:
        """Get projects accessible by this user (user known by api-key).

        Args:
            permission: CKAN permission this user should have in order to include
                        the project (aka organization) in the list. Other options
                        include: "read", "manage_group"
            fields: List of fields from the project (organization) record to return.
                    Default is "id". Others include "name", "created", "description".
                    You can explicitly pass `None` to include all fields.

        Returns:
            Dictionary where the key is the project's "display" name, which the
            user would see on the web page, and the value is a dict with keys equal
            to the list given to the `fields` argument (and values from the record).
        """
        try:
            r = requests.post(
                self._action_url("organization_list_for_user"),
                json={"permission": permission},
                headers=self._apikey_header(),
                verify=True,
            )
        except requests.ConnectionError as err:
            raise CKANError(f"Cannot connect to {self._url} [{err}]")
        rjson = r.json()
        if not rjson["success"]:
            raise CKANError(f"Cannot get list of organizations: " f"{rjson['error']}")
        info = {}
        for proj_info in rjson["result"]:
            d = {}
            if fields is None:  # None means "all"
                d = proj_info
            elif len(fields) == 0:  # Pick default field(s)
                d = {"id": proj_info["id"]}
            else:
                d = {f: proj_info[f] for f in fields}
            info[proj_info["display_name"]] = d
        return info


    def _action_url(self, action):
        return f"{self._url}/api/3/action/{action}"

    def _download_url(self, dataset_id):
        return f"{self._url}/dataset/{dataset_id}"

    def _apikey_header(self):
        return {"X-CKAN-API-Key": self._api_key}

    def _apikey_auth_header(self):
        return {"Authorization": self._api_key}

[docs]    def upload(
        self,
        name: str,
        records=None,
        files: List[str] = None,
        basedir=".",
        use_existing: bool = True,
    ) -> str:
        """Upload all the records and files into a new CKAN "package".

        Args:
            name: Name of the CKAN package
            records: The records to upload
            files: The files to upload
            basedir: Base directory for all the files
            use_existing: Do not create package
        Returns:
            Package id of created (or existing) package.
        Raises:
            CKANError: if something goes wrong on the CKAN end.
        """
        self._dsinfo["name"] = name
        create_kw = {}
        if self._proxies:
            create_kw["proxies"] = self._proxies
        if not use_existing:
            result = self._create_package(**create_kw)
            pkgid = result["name"]
        else:
            pkgid = name
        if records:
            self._upload_records(pkgid, records)
        if files:
            self._upload_files(pkgid, basedir, files)
        return pkgid


    def _create_package(self, **kwargs):
        self.notify(f"create package '{self._dsinfo['name']}'")
        r = requests.post(
            self._action_url("package_create"),
            json=self._dsinfo,
            headers={
                "X-CKAN-API-Key": self._api_key,
                "content-type": "application/json;" "charset=utf-8",
            },
            verify=True,
            **kwargs,
        )
        response = r.json()
        if not response["success"]:
            if "in use" in response["error"]["name"][0]:
                raise CKANPackageExistsError(f"Package exists: {response['error']}")
            else:
                raise CKANError(f"Creating package: {response['error']}")
        return response["result"]

    def _get_package_resources(self, ds_id):
        url_base = self._action_url("package_show") + "?id="
        headers = self._apikey_auth_header()
        _log.debug(f"requests.get url='{url_base + ds_id}")
        response = requests.get(url_base + ds_id, headers=headers)
        rjson = response.json()
        if not rjson["success"]:
            msg = rjson["error"]["message"]
            raise CKANError(f"Cannot get package: {msg}")
        return rjson["result"]["resources"]

    def _upload_records(self, pkgid, records):
        with TemporaryDirectory() as tmpdir:
            i = 1
            for rec in records:
                name = f"record{i}_json"
                self.notify(f"record {name}")
                # upload JSON
                meta = {
                    "name": name,
                    "title": f"Record {i}",
                    "mimetype": "application/json",
                    "format": "JSON",
                    "package_id": pkgid,
                }
                path = Path(tmpdir) / f"record{i}.json"
                rec.as_json(path.open("w"))
                _log.debug(f"upload to CKAN: {path}")
                self._upload(meta, path, mode="r")
                # upload CSV file for each dataset
                meta = {"mimetype": "text/csv", "format": "CSV", "package_id": pkgid}
                j = 1
                for dset in rec.data.datasets:
                    if hasattr(dset, "filenames"):
                        # upload list of existing files
                        for filename in dset.filenames:
                            meta["name"] = filename
                            meta["title"] = f"Data file {filename}"
                            path = Path(filename)
                            self._upload(meta, path, mode="r")
                    else:
                        # upload inline data, after first exporting it to CSV
                        meta["name"] = f"record{i}_table{j}"
                        meta["title"] = (f"CSV data {j} in Record {i}",)
                        path = Path(tmpdir) / f"record{i}_{j}.csv"
                        with path.open(mode="w", newline="") as csvfile:
                            writer = csv.writer(csvfile)
                            writer.writerow(
                                [
                                    f"{name} ({units})"
                                    for name, units in dset.header.columns
                                ]
                            )
                            for row in dset.observations:
                                writer.writerow(row)
                        self._upload(meta, path, mode="r")
                    j += 1
                i += 1

    def _upload_files(self, pkgid, basedir, files):
        for filename in files:
            if filename.endswith(".ipynb"):
                mtype = "application/x-ipynb+json"
                mformat = "JSON"
            else:
                (mtype, enc) = mimetypes.guess_type(filename)
                if mtype is None:
                    _log.warning(
                        f"Cannot determine mimetype for {filename}: using "
                        f"application/octet-stream"
                    )
                    mtype = "application/octet-stream"  # generic
                mformat = mtype.split("/")[1].upper()
            path = Path(basedir) / filename
            name = path.parts[-1]
            self.notify(f"file '{path}'")
            meta = {
                "name": name,
                "title": path.parts[-1],
                "package_id": pkgid,
                "mimetype": mtype,
                "format": mformat,
            }
            self._upload(meta, path, mode="rb")

    def _upload(self, meta: dict, path: str, mode="r"):
        """Upload files into CKAN dataset.
        """
        _log.debug(
            f"CKAN._upload: data={meta}, file={path} headers={self._apikey_header()}"
        )
        r = requests.post(
            self._action_url("resource_create"),
            data=meta,
            headers=self._apikey_header(),
            verify=True,
            files=[("upload", open(path, mode=mode))],
        )
        response = r.json()
        if not response["success"]:
            raise CKANError(f"Uploading record: {response['error']}")


[docs]    def download(
        self,
        project: str = None,
        dataset: str = None,
        dest_dir: str = None,
        overwrite: str = None,
    ) -> (int, int):
        """Download all resources from project + dataset into dest_dir.

        Args:
            project: Project name
            dataset: Dataset ID
            dest_dir: Destination directory name. Assume it
                      already exists
            overwrite: Action if file would be overwritten:
               - None: Do nothing, overwrite file
               - "error": Raise an error. Don't download anything.
        Returns:
            (num_files, num_bytes) downloaded
        Raises:
            CKANError: Connecting/downloading from CKAN
            KeyError: If there was a duplicate filename, and
                      overwrite_ok was False.
        """
        resources = self._get_package_resources(dataset)
        # download the files in the dataset
        dest_path = Path(dest_dir)

        # check first for duplicate/overwrite
        if not overwrite == "error":
            for rsrc in resources:
                fname = rsrc["url"].rsplit("/", 1)[-1]
                if (dest_path / fname).exists():
                    raise KeyError(str(fname))
        # download the data
        num_files, num_bytes = 0, 0
        for rsrc in resources:
            url = rsrc["url"]
            fname = url.rsplit("/", 1)[-1]
            num_bytes += self._fetch_resource(url, dest_path / fname)
            num_files += 1
        return num_files, num_bytes


    def _fetch_resource(self, url, path):
        r = requests.get(url, headers=self._apikey_auth_header())
        with path.open("w") as fp:
            nbytes = fp.write(r.text)
        return nbytes
