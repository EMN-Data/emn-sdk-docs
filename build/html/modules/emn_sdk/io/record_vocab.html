[docs]class Vocabulary:
    """For the EMN record metadata, provide ability to add a vocabulary of
    required fields, conditional on the presence of certain other
    field/value combinations. This allows us to basically treat some fields as
    specifying the "type" of record, and then check that other
    required fields are present for records of that type.

    Note to developers: this uses the "if"/"then" features in JSON Schema Draft 7.
    """

    TBL_COLS = ("level", "field", "value", "require")

    def __init__(self, spec: Dict):
        """Create from a specification.

        Args:
            spec: Vocabulary specification. The specification is a dictionary of
                  the following form::

                    { "lab"|"experiment"|"measurement"|"sample" :
                      [ <ITEM>, <ITEM>, ... ] }

                where each `ITEM` is of the form::

                    { "if": ("<FIELD>", "<VALUE>"),
                      "require": [ <PROP>, <PROP>, .. ] }

                and, finally, each `PROP` is either a string name of a field OR
                a tuple of length 2 where the first item is the name of the field
                and the second is a dictionary of JSON-Schema constraints such as
                "pattern", "type", "enum", "minimum", "maximum", etc.

                In plain English, what this is accomplishing is to specify:

                * For a metadata section S,
                * `if` the value of a field X is some known value Y,
                * then `require` that other fields Z(1), Z(2), ..., Z(n)
                  are present as well.
                * Optionally, also require that the Z fields are
                  certain types and/or values.
                * If 'Y' is empty, then this will apply if 'X' is present at all
                * if 'X' is empty, then 'Z(n)' will be required of all metadata
        """
        self._spec = spec

    @property
    def specification(self):
        return self._spec


[docs]    def extend(self, schema):
        working_schema = copy.deepcopy(schema)
        return self._add_conditions(working_schema)


    def _add_conditions(self, schema):
        # loop over each 'level' (sub-section) of metadata
        for lvl in self._spec:
            # metadata subschema definition
            md_sect = schema["definitions"][f"{lvl}-metadata"]
            # section with existing list of conditions
            item_vals = md_sect["items"]["properties"]["values"]
            allof = item_vals.get("allOf", None)
            if allof is None:
                allof = item_vals["allOf"] = []
            # create new conditions
            conditions = []
            for cond in self._spec[lvl]:
                if_name, if_value = cond["if"]
                # empty name means just extend properties
                if if_name == "":
                    for prop in cond["require"]:
                        item_vals["properties"][prop] = {}
                    item_vals["required"].extend(cond["require"])
                    continue
                # otherwise add a new condition
                props, props_req = {}, []
                for prop in cond["require"]:
                    if isinstance(prop, str):
                        props[prop] = {}
                        props_req.append(prop)
                    elif len(prop) == 2:  # tuple name, subschema
                        name, subschema = prop
                        props[name] = subschema
                        props_req.append(name)
                # if the if_value is empty, build a full expression but
                # don't restrict the possible values
                if if_value == "":
                    value_expr = {}
                else:
                    value_expr = {"const": if_value}
                c = {
                    "if": {"properties": {if_name: value_expr}},
                    "then": {"properties": props, "required": props_req},
                }
                conditions.append(c)
            # add conditions
            allof.extend(conditions)
        return schema

    def from_csv(cls, obj: Union[str, TextIO]) -> "Vocabulary":
        if isinstance(obj, str):
            # CSV filename or CSV data in a buffer
            if len(obj) < 100 and obj.endswith(".csv"):
                data = csv.reader(open(obj))
            else:
                data = csv.reader(StringIO(initial_value=obj))
        elif hasattr(obj, "read"):
            # CSV file object
            data = csv.reader(obj)
        else:
            raise TypeError(f"String or file-like object expected, got: {type(obj)}")
        spec = cls._spec_from_table(data)
        return cls(spec)


    @classmethod
    def _spec_from_table(cls, data) -> Dict:
        """Expected columns in table:

        level | field | value | require

        The `require` field may be a field, a comma-separated list of
        fields, or a complex .

        Any other table columns are ignored.

        Returns:
            Specification as expected by `Vocabulary` constructor.
        Raises:
            ValueError if the table format is bad
        """
        spec = {}
        cols = {c: None for c in cls.TBL_COLS}
        for i, row in enumerate(data):
            if i == 0:
                # first row is header. find columns (or error)
                for j, col in enumerate(row):
                    col = col.lower()
                    if col == "required":  # allow this alias
                        col = "require"
                    if col in cols:
                        cols[col] = j
                if None in cols.values():
                    missing = [field for field in cols if cols[field] is None]
                    raise ValueError(f"Missing columns: {', '.join(missing)}")
            else:
                # every other row is a requirement
                lvl = row[cols["level"]]
                if lvl not in spec:
                    spec[lvl] = []
                required = row[cols["require"]].strip()
                if ";" in required:  # complex form (name, subschema)
                    name, subsch = required.split(";")
                    name = name.strip()
                    # treat subschema as a dict (without outer braces)
                    subsch = "{" + subsch.strip().replace("'", '"') + "}"
                    required = [(name, json.loads(subsch))]
                elif "," in required:
                    required = [r.strip() for r in required.split(",")]
                else:
                    required = [required]
                match_name, match_value = row[cols["field"]], row[cols["value"]]
                # allow some aliases for empty values
                # also force value to be empty in that case
                if match_name in ("-", "n/a", "N/A"):
                    match_name, match_value = "", ""
                expr = {"if": (match_name, match_value), "require": required}
                # add/merge with spec
                merged = False
                for spec_expr in spec[lvl]:
                    if spec_expr["if"] == expr["if"]:
                        spec_expr["require"].extend(expr["require"])
                        merged = True
                        break
                if not merged:
                    spec[lvl].append(expr)
        return spec

    def from_ckan(cls, user: str, passwd: str, dataset_id: str, *args) -> "Vocabulary":
        """Create from CKAN dataset.

        Args:
            user: CKAN login name
            passwd: CKAN login password
            dataset_id: CKAN dataset (package) with vocabulary resource(s)
            *args: Passed to :class:`io.ckan.CKANBase` constructor

        Returns:
            Instance, initialized from the dataset.

        Raises:
            CKANError: for problems with CKAN
            ValueError: for problems with the data itself
        """
        ckan_vocab = CKANVocabulary(*args)
        vocab = ckan_vocab.fetch(user, passwd, dataset_id=dataset_id)
        return vocab

[docs]    def fetch(
        self,
        user: str,
        passwd: str,
        dataset_id: str = None,
        ignore_other_csv: bool = True,
    ) -> Vocabulary:
        """Fetch vocabulary from combined resources in the given dataset.

        Args:
            user: CKAN login name
            passwd: CKAN password
            dataset_id: CKAN dataset ID
            ignore_other_csv: If True, allow non-vocabulary CSV, otherwise
                              raise a ValueError.
        Returns:
            Vocabulary instance
        Raises:
            ValueError: if there are no (parse-able) resources with data
            CKANError: problems communicating with CKAN
        """
        self._bad_csv = ignore_other_csv
        # find and retrieve the data
        if dataset_id is None:
            raise ValueError("Vocabulary dataset ID is required")
        try:
            table = self._fetch_data(dataset_id, user, passwd)
        except CKANError as err:
            raise
        if len(table) == 1:  # only header row means no data
            bad = " valid " if ignore_other_csv else ""
            raise ValueError(f"No{bad}vocabulary resources found in '{dataset_id}'")
        vocab = Vocabulary.from_table(table)
        return vocab


    def _login(self, username, password):
        """User/password login is needed to download data from CKAN,
        i.e. there is no API for it.

        Returns:
            Session with logged in user
        Raises:
            RuntimeError: if the login fails
        """
        s = requests.Session()
        data = {"login": username, "password": password}
        url = self._url + "/login_generic"
        r = s.post(url, data=data)
        if "field-login" in r.text:
            # Response still contains login form
            raise CKANError(f"login failed for user {username}")
        return s

    def _fetch_data(self, dsid, user, passwd):
        """Find and fetch data."""
        # find vocabulary resources in dataset
        show_info = {
            "owner_org": self.project,
            "institution": self.institution,
            "id": dsid,
        }
        r = requests.post(
            self._action_url("package_show"),
            json=show_info,
            headers=self._apikey_header(),
            verify=True,
        )
        rjson = r.json()
        if not rjson["success"]:
            raise CKANError(
                f"Cannot get resources in dataset {dsid}: " f"{rjson['error']}"
            )
        resources = rjson["result"]["resources"]

        # download CSV files and parse as tables
        sess = self._login(user, passwd)
        table = [list(Vocabulary.TBL_COLS)]  # new, with header
        for rsrc in resources:
            # only look at CSV files (use mimetype; more reliable than 'format')
            if rsrc["mimetype"] != "text/csv":
                continue
            # get the data from the CKAN session
            url = rsrc["url"]
            r = sess.get(url)
            if r.status_code != 200:
                try:
                    rjson = r.json()
                except json.JSONDecodeError:
                    rjson = {"error": str(r.status_code)}
                raise CKANError(f"Getting resource {url}: {rjson['error']}")
            # convert CSV data into a table
            try:
                rsrc_table = csv.reader(StringIO(initial_value=r.content.decode()))
            except UnicodeDecodeError as err:
                if self._bad_csv:
                    continue
                else:
                    raise ValueError(
                        f"Error decoding '{rsrc['name']}' "
                        f"in dataset '{dsid}': {err}"
                    )
            # get index mapping of columns in the input table
            # note: Create a table with only required columns, in a standard order.
            #       This allows all the tables to be simply concatenated.
            hdr = next(rsrc_table)
            try:
                col_idx = [hdr.index(c) for c in table[0]]
            except ValueError as err:
                if self._bad_csv:
                    col_idx = None
                else:
                    bad_key = str(err).split()[0]
                    raise ValueError(
                        f"Error parsing resource '{rsrc['name']}' "
                        f"in dataset '{dsid}': missing {bad_key} column"
                    )
            # add to main table (using index mapping)
            if col_idx:
                for row in rsrc_table:
                    table.append([row[i] for i in col_idx])
        return table
