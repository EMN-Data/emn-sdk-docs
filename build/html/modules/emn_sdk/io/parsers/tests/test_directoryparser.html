[docs]def test_conflicting(tmp_path):
    # cannot have 2 parser instances match the same file
    p = directory_parser.Parser({"*": DummyParser(), "*.txt": DummyParser()})
    afile = tmp_path / "foo.txt"
    fp = afile.open("w")
    fp.close()
    pytest.raises(KeyError, p.parse, tmp_path)
    # this is ok, since the parser is the same instance
    dp = DummyParser()
    p = directory_parser.Parser({"*": dp, "*.txt": dp})
    p.parse(tmp_path)
    # this is also ok, since there's only one parser
    dp = DummyParser()
    p = directory_parser.Parser({"*.txt": DummyParser()})
    p.parse(tmp_path)

[docs]def test_basic(tmp_path):
    add_files(tmp_path, 3)
    dummy = DummyParser()
    mapping = {"*.txt": dummy}
    p = directory_parser.Parser(mapping)
    records = p.parse(str(tmp_path))
    assert len(records) == 3
    _log.debug(f"records for *.txt = {records}")
    for filename in records.keys():
        assert filename.endswith(".txt")
