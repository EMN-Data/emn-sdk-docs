[docs]def timed(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        name = func.__name__
        dur = round(end - start, 3)
        if name not in _total_times:
            _total_times[name] = 0
        _total_times[name] += dur
        _log.debug(
            f"fn={name:20s} sec={dur:.3f} "
            f"ttl_sec={round(_total_times[name], 3):.3f}"
        )
        return result

    return wrapper




[docs]def get_epoch_time(datestr):
    """Convert a date - time string to seconds elapsed since
        Jan 1, 1970 00:00:00 UTC and a formatted date string

    Args:
        datestr: Input time string

    Returns:
       zerpoint time (epoch time of first data point) and collection date
    """
    zp = datestr.split()
    if len(zp[1]) < 2:
        zp[1] = "0" + zp[1]
    if zp[4] == "PM":
        ms_time = 0
    else:
        ms_time = 1  # either 24H time or AM

    # convert to struct_time argument
    if ms_time == 0:
        strc_time = time.strptime(
            zp[0] + " " + zp[1] + " " + zp[2] + " " + zp[3] + zp[4],
            "%b %d %Y %I:%M:%S%p",
        )
    else:
        strc_time = time.strptime(
            zp[0] + " " + zp[1] + " " + zp[2] + " " + zp[3], "%b %d %Y %H:%M:%S",
        )

    # convert to seconds elapsed since Jan 1, 1970 00:00:00 UTC
    etime = time.mktime(strc_time)
    # convert to formatted date string
    cdate = time.strftime("%b %d %Y", strc_time)
    return etime, cdate




[docs]def interp_data(xvals: List[float], data: pd.DataFrame):
    """Interpolate data to match the mass spec time stamps.

    Args:
        xvals: mass spec time in seconds
        data: data to interpolate
    Returns:
        (dict): keys are settings/column names; values are 1D arrays
    """
    interp_data = dict()
    interp_data["time"] = np.array(xvals)
    dxvals = np.array(data["time"])
    for col in data:
        if col != "time":
            interp_data[col] = np.interp(xvals, dxvals, data[col])
    return interp_data




[docs]def merge_dict(data):
    """Merge multiple dicts and DataTables.

    Args:
        data: data streams to convert
    Returns:
        datadict (dict): merged dictionary with one time column
    """
    datadict = dict()
    for d in data:
        for col in d:
            if col in datadict and col != "time":
                datadict[col + " Setting"] = d[col].tolist()
            elif col not in datadict:
                datadict[col] = d[col].tolist()
    return datadict

def create_exp_data(path: Path, exp_name, zeroPointTime):
    """Convert data arrays to a datatable.

    Args:
        zeroPointTime (int): start time of the experiment in absolute seconds
    Returns:
        (pandas DataFrame): system reading data
    """
    exp_data = pd.read_csv(str(path / exp_name), sep="\t")

    # convert LabVIEW epochTime to UNIX epochTime
    # 24107 days between Jan 1, 1904 and Jan 1, 1970
    exp_data["time"] = exp_data["time"] - (24107 * 86400)
    return exp_data

def create_stat_data(path: Path, exp_num, time_arr, collect_date):
    """Extract additional system settings to a datatable.

    Args:
        epoch time array and collection date of the experiment
    Returns:
        stat_dict and meta data dict
    """
    stat_dict = {"time": time_arr.to_numpy(), "Laser": [], "BYPASS": [], "TRAP": []}
    laser_data = []
    bypass_data = []
    trap_data = []
    line_dict = None

    # go through the STAT file to determine relevant lines.
    # append boolean "states" as values to those lists
    for line in (path / f"STAT{exp_num}.txt").open():
        line_dict = json.loads(line)

        if "TYPE" in line_dict:
            # parse time coordinate, convert to epoch time
            status_time_array = line_dict["time"].split(".")
            status_time = get_epoch_time(
                collect_date + " " + status_time_array[0] + " M"
            )[0]
            status_time += float(status_time_array[-1]) / 10

            # update values in Data lists
            value = 0
            if "Laser" in line_dict["LABEL"]:
                if bool(line_dict["NewVal"]["Laser State"]) is True:
                    value = 1
                laser_data.append([status_time, value])
            elif "BYPASS" in line_dict["LABEL"]:
                if bool(line_dict["NewVal"]) is True:
                    value = 1
                bypass_data.append([status_time, value])
            elif "TRAP" in line_dict["LABEL"]:
                if bool(line_dict["NewVal"]) is True:
                    value = 1
                trap_data.append([status_time, value])

    # the LAST line read contains all metadata dicts
    all_meta = line_dict

    # pop elements from lists if they exist, otherwise set value to 0 for
    # all time.
    # also set initial states to tfirst if neccessary
    tfirst = stat_dict["time"][0]
    tlast = stat_dict["time"][-1]

    if len(laser_data) >= 2:
        now_laser = laser_data.pop(0)
        if now_laser[0] < tfirst:
            now_laser[0] = tfirst
        next_laser = laser_data[0]
    elif len(laser_data) == 1:
        now_laser = laser_data.pop(0)
        if now_laser[0] < tfirst:
            now_laser[0] = tfirst
        next_laser = [tlast + 1, now_laser[1]]
    else:
        now_laser = [tlast + 1, 0]
        next_laser = now_laser

    if len(bypass_data) >= 2:
        now_bypass = bypass_data.pop(0)
        if now_bypass[0] < tfirst:
            now_bypass[0] = tfirst
        next_bypass = bypass_data[0]
    elif len(bypass_data) == 1:
        now_bypass = bypass_data.pop(0)
        if now_bypass[0] < tfirst:
            now_bypass[0] = tfirst
        next_bypass = [tlast + 1, now_bypass[1]]
    else:
        now_bypass = [tlast + 1, 0]
        next_bypass = now_bypass

    if len(trap_data) >= 2:
        now_trap = trap_data.pop(0)
        if now_trap[0] < tfirst:
            now_trap[0] = tfirst
        next_trap = trap_data[0]
    elif len(trap_data) == 1:
        now_trap = trap_data.pop(0)
        if now_trap[0] < tfirst:
            now_trap[0] = tfirst
        next_trap = [tlast + 1, now_trap[1]]
    else:
        now_trap = [tlast + 1, 0]
        next_trap = now_trap

    # create stat_dict by mapping state changes onto experiment epoch time
    las = []
    byp = []
    tra = []
    for t in stat_dict["time"]:
        las.append(now_laser[1])
        byp.append(now_bypass[1])
        tra.append(now_trap[1])

        if t >= next_laser[0] and len(laser_data) > 0:
            now_laser = laser_data.pop(0)
            if len(laser_data) > 0:
                next_laser = laser_data[0]
            else:
                next_laser = [tlast + 1, now_laser[1]]

        if t >= next_bypass[0] and len(bypass_data) > 0:
            now_bypass = bypass_data.pop(0)
            if len(bypass_data) > 0:
                next_bypass = bypass_data[0]
            else:
                next_bypass = [tlast + 1, now_bypass[1]]

        if t >= next_trap[0] and len(trap_data) > 0:
            now_trap = trap_data.pop(0)
            if len(trap_data) > 0:
                next_trap = trap_data[0]
            else:
                next_trap = [tlast + 1, now_trap[1]]

    # need to convert to numpy arrays for code legacy issues
    stat_dict["Laser"] = np.array(las)
    stat_dict["BYPASS"] = np.array(byp)
    stat_dict["TRAP"] = np.array(tra)
    return stat_dict, all_meta

def create_ms_data(path: Path, num):
    """Read mass spectrometry data from Merlin TXT file into a datatable

    Returns:
        msData (pandas DataFrame): the mass spectrometry data
        zeroPointTime: start of MS collection
        date of MS collection
    """
    mass_to_compound = MASS_TO_COMPOUND
    # csvfile = csv.reader(open("data/massToCompound.csv"))
    # mass_to_compound = dict(csvfile)

    # this method is rather crude, but since the file won't get too big ever,
    # this shouldn't be a problem
    # BUT error prone due to changes in Merlin file formats!!

    compound_masses = []
    with open(str(path / "Mdata") + num + ".log") as msLog:
        # extract date/time from the second line of Mdata.log files
        line = msLog.readline()
        line = msLog.readline()
        epoch_time, collection_date = get_epoch_time(line[2:])

        # extract average masses of compounds from getting the lines after the
        # Q1FM Header
        while "Q1FM" not in line:
            line = msLog.readline()
        line = msLog.readline()
        test_array = []
        while "Tune Tables" not in line:
            if len(line) > 1:
                test_array.append(line.split())
                compound_masses.append(
                    (float(line.split()[0]) + float(line.split()[1])) / 2
                )
            line = msLog.readline()

    # match common masses to names using a dictionary saved as a csv file
    column_names = []
    for mass in compound_masses:
        column_names.append(mass_to_compound.get(str(round(mass)), str(round(mass))))

    column_names.append("time")

    # read mass spec data with proper column names
    index_col = len(column_names) - 1
    ms_data = pd.read_csv(
        str(path / "Mdata") + num + ".txt",
        dtype=float,
        index_col=index_col,
        delim_whitespace=True,
        names=column_names,
    )

    # change time array to absolute epochTime
    ms_data["time"] = 60.0 * ms_data["time"] + epoch_time
    zero_point_time = ms_data["time"].iloc[0]
    return [ms_data, zero_point_time, collection_date]

    def match(directory):
        """Match a directory of STCH data.

        Expect the following glob patterns to be present:

        - Mdata??.txt
        - DAC??.txt
        - POX??.txt
        - STAT??.txt

        There should be the same number of each file type (pattern).
        """
        p = Path(directory)
        files_of_type = {}
        for base_name in "Mdata", "DAC", "POX", "STAT":
            files_of_type[base_name] = len(list(p.glob(f"{base_name}??.txt")))
        num_mdata = files_of_type["Mdata"]
        if num_mdata > 0 and all((n == num_mdata for n in files_of_type.values())):
            return True
        return False

[docs]    def parse(self, path: Union[Path, str]):
        """Convert raw data and return as records.
        """
        if isinstance(path, str):
            path = Path(path)
        # get num_str from file base names in directory using Mdata head
        ms_file_nums = []
        for file in glob.glob(str(path / "Mdata*.txt")):
            line = os.path.split(file)[-1]
            ms_file_nums.append(line.split("Mdata")[-1][:2])

        if not ms_file_nums:
            raise ValueError(f"No Mdata*.txt files found in {path}")
        all_meta = None  # will be set in loop below

        # dict storing the run number and data dictionary as the key/value
        total_data_dict = dict()
        rec = Record()

        # find and process relevant files for each experiment
        uuids = {}
        for dataset_num in ms_file_nums:
            ms_result = create_ms_data(path, dataset_num)
            pox_result = create_exp_data(path, f"POX{dataset_num}.txt", ms_result[1])
            dac_result = create_exp_data(path, f"DAC{dataset_num}.txt", ms_result[1])
            pox_dict = interp_data(ms_result[0]["time"], pox_result)
            dac_dict = interp_data(ms_result[0]["time"], dac_result)
            stat_dict, all_meta = create_stat_data(
                path, dataset_num, ms_result[0]["time"], ms_result[2]
            )
            all_meta["experiment_MD"]["file_index"] = dataset_num
            if not uuids:
                uuids = rec.meta.add(
                    {
                        rec.meta.LVL_LAB: all_meta["dataset_MD"],
                        rec.meta.LVL_EXP: all_meta["lab_experimental_MD"],
                        rec.meta.LVL_SMP: all_meta["sample_MD"],
                    }
                )
            total_data_dict[dataset_num] = dict()
            # total_data_dict[num_str]["rawdat"] = merge_dict(
            #    [ms_result[0], pox_dict, dac_dict, stat_dict]
            # )
            all_data = merge_dict([ms_result[0], pox_dict, dac_dict, stat_dict])
            # add measurement metadata
            uuids[rec.meta.LVL_MSR] = rec.meta.add_one(
                rec.meta.LVL_MSR, all_meta["experiment_MD"]
            )
            # add measurement data
            if self._inline:
                dataset = self._add_data_inline(dataset_num, all_data)
            else:
                dataset = self._add_data_file(dataset_num, all_data)
            dataset.set_meta(rec.meta, uuids.copy())
            rec.data.add_dataset(dataset)

            # total_data_dict[num_str]["exmeta"] = all_meta["experiment_MD"]

        # total_data_dict["dsmeta"] = all_meta["dataset_MD"]
        # total_data_dict["lameta"] = all_meta["lab_experimental_MD"]
        # total_data_dict["sameta"] = all_meta["sample_MD"]
        # return total_data_dict

        return [rec]


    @staticmethod
    def _add_data_inline(dataset_num, all_data):
        columns = list(all_data.keys())
        data_hdr = DatasetHeader()
        for col in columns:
            data_hdr.add_column(col, "")
        ds = Dataset(name=dataset_num)
        ds.set_header(data_hdr)
        for i in range(len(all_data[columns[0]])):
            row = [all_data[col][i] for col in columns]
            ds.add_observation(row)
        return ds

    def _add_data_file(self, dataset_num, all_data):
        import csv

        # create csv file
        filename = self.DATA_FILE_NAME.format(num=dataset_num)
        with open(filename, "w") as csv_file:
            w = csv.writer(csv_file, dialect="emn")
            columns = list(all_data.keys())
            w.writerow(columns)  # header row
            for i in range(len(all_data[columns[0]])):
                row = [all_data[col][i] for col in columns]
                w.writerow(row)
        ds = RawDataset()
        ds.add_file(filename)
        return ds
