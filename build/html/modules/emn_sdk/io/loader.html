[docs]    @classmethod
    def from_toml(cls, str_or_file, **kwargs):
        """Construct from TOML string or fiile.

        Expected TOML contents::

            [CKAN]
            organization = "API Sandbox"
            institution = "Lawrence Berkeley National Laboratory"

        Args:
            str_or_file (Union[str,file]): Input. If it has a 'read'
                method, it is assumed to be a file. Otherwise it is
                assumed to be a string, after checking that it is not
                a filename.
            kwargs: Additional key/value pairs

        Returns:
            DatasetInfo: Constructed object.

        """
        if hasattr(str_or_file, 'read'):
            # it is a file
            conf = toml.load(str_or_file)
        elif not '\n' in str_or_file and not '[' in str_or_file:
            # it is not TOML data, so try it as a filename
            conf = toml.load(open(str_or_file))
        else:
            # assume it is a string
            conf = toml.loads(str_or_file)
        info = conf['CKAN']
        info.update(kwargs)
        return cls(**info)

[docs]    def upload_files(self, files, **kwargs):
        """Upload files to the dataset.

        Calls :meth:`Archive.upload_files()` method, with this
        dataset as the `target`, on the :class:`Archive` instance
        used to create this object.

        Args:
            files (List[str]): Filenames (or single filename)
            **kwargs: Arguments passed to :meth:`Archive.upload_files()`.

        Returns:

        """
        if not self._archive:
            raise ValueError('No archive specified when DatasetResult object '
                             'was created. Try using Archive.upload()')
        return self._archive.upload_files(self, filenames=files, **kwargs)

[docs]    def create_dataset(self, dataset_info):
        """Create a new dataset in CKAN for the provided dataset info.

        Args:
            dataset_info (DatasetInfo): Information about the dataset

        Returns:
            DatasetResult: Information about created dataset.
        """
        kwargs = self._post_kw.copy()
        r = requests.post(self._action_url('package_create'),
                          json=dataset_info.json,
                          headers={'X-CKAN-API-Key': self._apikey,
                                   'content-type': 'application/json;'
                                                   'charset=utf-8'},
                          verify=True,
                          **kwargs)
        response = r.json()
        success = response['success']
        if not success:
            error = response['error']
            print(error)
            if error['name'] == 'That URL is already in use.':
                raise DuplicateDatasetError(dataset_info)
            else:
                raise CreateDatasetError(dataset_info, why=error['name'])
        return DatasetResult(response['result'], archive=self)



[docs]    def upload_files(self, target, filenames=None, metadata=None):
        """Upload file resources into the target dataset.

        Args:
            target (DatasetResult): Target dataset
            filenames (List[str]): List of filenames (paths).
            metadata (dict): Other key/value pairs added to resource metadata
        Returns:
            None

        """
        kwargs = self._post_kw.copy()
        if filenames:
            if isinstance(filenames, str):
                filenames = [filenames]
        meta = {
            'package_id': target.json['name'],
            'mimetype': 'application/json',
            'format': 'JSON'
        }
        if metadata:
            meta.update(metadata)
        for fname in filenames:
            meta['name'] = fname
            meta['name'] = meta['name'].split("/").pop()
            meta['title'] = os.path.basename(fname)
            # print('@@ JSON meta: {}'.format(meta))
            r = requests.post(self._action_url('resource_create'),
                              data=meta,
                              headers={'X-CKAN-API-Key': self._apikey},
                              verify=True,
                              files=[('upload', open(fname, 'rb'))],
                              **kwargs)
            response = r.json()
            success = response['success']
            if not success:
                raise UploadError(response['error'])

            # If there is embedded data, upload that as CSV
            # TODO: see if this still runs with rdata.json
            contents = json.load(open(fname, 'rb'))
            if 'data' in contents and len(contents['data']['data']) > 0:
                self._upload_csv(contents, meta, fname, kwargs)


    def _upload_csv(self, contents, meta, fname, kwargs):
        data = contents['data']['data']  # less confusing name
        ncols = len(data)
        nrows = len(data[0])
        # construct CSV file name from base file name
        csv_fname = '{}.csv'.format(os.path.splitext(
            os.path.basename(fname))[0])
        csv_strio = io.StringIO()
        csv_out = csv.writer(csv_strio)
        for i in range(nrows):
            row = [data[j][i] for j in range(ncols)]
            csv_out.writerow(row)
        csv_strio.seek(0)  # reset for reading back
        meta['name'] = 'data'
        meta['title'] = 'Data Table'
        meta['format'] = 'CSV'
        meta['mimetype'] = 'text/csv'
        # print('@@ CSV meta: {}'.format(meta))
        r = requests.post(self._action_url('resource_create'),
                          data=meta,
                          headers={'X-CKAN-API-Key': self._apikey},
                          verify=True,
                          files=[('upload', csv_strio)],
                          **kwargs)
        response = r.json()
        success = response['success']
        if not success:
            raise UploadError('Uploading CSV data: {}'.format(
                response['error']))
