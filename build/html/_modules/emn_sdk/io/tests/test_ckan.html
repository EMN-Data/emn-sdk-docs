[docs]def test_get_projects_fields(ckan_info: CKANInfo):
    if ckan_info.url and ckan_info.apikey:
        c = ckan.CKANBase(ckan_info.url, ckan_info.apikey)
        # user-supplied fields
        proj = c.get_projects(fields=["name", "created", "description"])
        assert len(proj) > 0
        for key, value in proj.items():
            assert len(key) > 0
            assert "name" in value
            assert "created" in value
            assert "description" in value
            assert "title" not in value
        # default field (id)
        proj = c.get_projects()
        assert len(proj) > 0
        for key, value in proj.items():
            assert "id" in value
            assert "name" not in value
        # ALL fields
        proj = c.get_projects(fields=None)
        assert len(proj) > 0
        for key, value in proj.items():
            assert len(key) > 0
            assert "name" in value
            assert "created" in value
            assert "description" in value
            assert "title" in value
    else:
        pytest.skip("CKAN_URL and CKAN_APIKEY needed")
