[docs]    def parse(self, path, require_files=True):
        """Parse a JSON file.

        Args:
            path: Path to input JSON file.
            require_files: If True (default), files mentioned in the JSON must exist.
                           Otherwise this will not be checked.
        Raises:
            IOError: If `require_files` is True and any of the files is missing.
        """
        rec = model.Record.from_json(path, require_files=require_files)
        return [rec]
