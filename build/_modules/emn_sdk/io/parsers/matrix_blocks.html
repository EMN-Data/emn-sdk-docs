"""
Parse a matrix of values into metadata and data block(s).
The values are fed into the algorithm one row at a time,
as, e.g., from a CSV file.

This is a utility module that will be used by spreadsheet/csv parsers.
"""
import logging

[docs]class MatrixParser:
    # recognizing a block header
    HEADER_RE = re.compile(
        r"\s*([Ll]ab|[Ee]xperiment|[Mm]easurement|[Ss]ample|[Ss]earch[Dd]ata|[Dd]ata)(:(\w|-_)+)?\s*$"
    )
    # recognizing a data block header
    DATA_HEADER_RE = re.compile(r"\s*[Dd]ata(:([\w\-_]+))?\s*$")
    # recognize a search data header
    SEARCH_DATA_RE = re.compile(r"\s*[Ss]earch[Dd]ata(:([\w\-_]+))?\s*$")

    def __init__(self):
        self._blocks = []
        self._merged = False
        self.row_num = -1
        self.prev_block = None
        self.width = -1


[docs]    def row(self, row):
        """Process next row of data.
        """
        self.row_num += 1
        i, j, n = self.row_num, 0, len(row)
        if self.prev_block is None:  # first row
            self.width = n
            self.prev_block = [None] * n
        elif n > len(self.prev_block):
            # pad, if shorter, so lookups always succeed
            self.prev_block.extend([None] * (n - len(self.prev_block)))
        while j < n:
            cell = row[j]
            empty = cell == ""
            # test whether cell above was in a block
            if self.prev_block[j] is None:  # NO, not in block
                # not in block, and no data -> do nothing
                if empty:
                    pass
                # not in block, and data -> maybe start a block
                else:
                    if self._block_header_key(cell):
                        if self._data_header_key(cell):
                            block = DataBlock(row, i, j)
                            self.prev_block[j] = block
                        else:
                            # Treat search-data and key/value blocks the same
                            if self._search_header_key(cell):
                                block = SearchDataBlock(row, i, j)
                            else:
                                block = KeyValueBlock(row, i, j)
                            self.prev_block[j] = block
                            # skip next cell, it's the other part of the header,
                            # and its value will be ignored
                            j += 1
                            self.prev_block[j] = block  # also mark as part of block
                        self._blocks.append(block)
                    else:
                        self.prev_block[j] = None  # nope, not a block
            else:  # YES, cell above was in a block
                block = self.prev_block[j]
                # in block, and empty -> end block
                if empty:
                    if block.incomplete():
                        raise MatrixParserError(
                            f"Parse error at row {i}, column {j}: "
                            f"Current block is not valid, but empty cell ends it"
                        )
                    # mark the columns as not in the block
                    for k in range(j, j + block.width):
                        self.prev_block[k] = None
                    # skip next cell(s), which are also part of finished block
                    j += block.width - 1
                    # close the block
                    block.close()
                # in block, and full -> add to block
                else:
                    if not block.active:
                        raise MatrixParserError(
                            f"Parse error at row {i}, column {j}: "
                            f"Current block is not active, but attempted to add cell"
                        )
                    # add columns from this row to the block
                    num_cols = block.add(row, i, j)
                    # skip ahead by number of columns added
                    j += num_cols - 1
            # next cell. note that 'skips' above take this into account.
            j += 1


    @property
    def blocks(self):
        """Return parsed blocks.
        """
        # just return the blocks if this was already done
        if self._merged:
            return self._blocks
        # create dict with one entry per block type
        b, num_merges = {}, 0
        for block in self._blocks:
            t = block.block_type
            if t in b:
                _log.debug(f"merging block type='{t}'")
                b[t].merge_data(block)
                num_merges += 1
            else:
                b[t] = block
        # recreate blocks from merged, if there were any merges
        if num_merges > 0:
            self._blocks = b.values()
        # set flag & return current blocks
        self._merged = True
        return self._blocks

    def _block_header_key(self, key):
        return self.HEADER_RE.match(str(key)) is not None

    def _data_header_key(self, key):
        return self.DATA_HEADER_RE.match(str(key)) is not None

    def _search_header_key(self, key):
        return self.SEARCH_DATA_RE.match(str(key)) is not None

[docs]class KeyValueBlock(Block):
    """A block for metadata, as key/value pairs.

    Expected format:
        row 1:    '<Type>[:<subtype>]', <ignored>
        row 2..N: <key>, <value>

    The ":subtype" is optional, and may be any alphanumeric
    sequence (with dash or underscores). The "type" must
    match a known metadata category. The value of <subtype>
    will be pre-pended to all the <key>s in the block.
    """

    TYPE_DELIM = "."

    def __init__(self, row, i, j):
        super().__init__(i, j)
        self.width = 2
        self._mtype, self._msubtype = None, None
        self._split_title(row[j])
        self._data.append([self._mtype, row[j + 1]])

    @property
    def block_type(self) -> str:
        return self._mtype


[docs]    def merge_data(self, block):
        """Merge block data. This removes the subtype.
        """
        self._data.extend(block.data)  # header already sliced out
        self._msubtype = None


    def _split_title(self, s):
        parts = s.split(":", 1)
        if len(parts) == 1:
            self._mtype = parts[0].lower()
        else:
            self._mtype, self._msubtype = parts[0].lower(), parts[1].lower()


[docs]    def add(self, row, i, j):
        if self._msubtype:
            self._data.append([self._msubtype + self.TYPE_DELIM + row[j], row[j + 1]])
        else:
            self._data.append(row[j : j + 2])
        return 2

[docs]class DataBlock(Block):
    """A block for data, which differs from key/value blocks.

    Expected format:
        row 1:     'Data[:<name>]', <other columns ignored>
        row 2:     '<unit>', '<unit>', .. , <blank>
        row 3..N:  value   , value   , ..

    The number of columns in the block is determined by how
    many non-empty "units" columns there are under the header (first) row.
    """

    def __init__(self, row, i, j):
        super().__init__(i, j)
        self._units = False
        # grab name of data block, if there is one
        header = row[j]
        m = MatrixParser.DATA_HEADER_RE.match(header)
        assert m is not None
        name = m.group(2)
        if name is None:
            _log.debug(f"DataBlock name=<empty>")
            self._name = ""
        else:
            _log.debug(f"DataBlock name='{name}'")
            self._name = name.lower()

    @property
    def block_type(self):
        """Append name to block type so that blocks with different
        names are not merged together (which would make no sense).
        """
        if self._name:
            return "data-" + self._name
        else:
            return "data"

    @property
    def name(self):
        return self._name


[docs]    def add(self, row, i, j):
        # if no units, then use this row as units.
        # This also sets the width
        if not self._units:
            n, k = len(row), j
            while k < n and row[k] != "":
                k += 1
            if k == j:
                raise MatrixParserError(
                    f"No 'units' data found in second row of data block, which"
                    f"starts at row {self._row_start}, column {self._col_start}"
                )
            self.width = k - j
            self._units = True
            # prepend a constructed header row
            self._data.append(["Data"] + [] * (self.width - 1))
        # in either case, append the row slice to the data
        self._data.append(row[j : j + self.width])
        # return width, or 1 if not known
        return self.width if self.width >= 0 else 1
