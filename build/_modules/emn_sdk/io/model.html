[docs]    def as_json(self, output_file: IOBase = None, **json_kw) -> Optional[str]:
        """

        Args:
            output_file: Output file (or None)
            **json_kw: Additional keywords passed to `json.dump[s]()`

        Returns:
            None if an output file is given.
            Otherwise a string of serialized JSON.
        """
        r = self._to_dict()
        # if no output file, return string
        if output_file is None:
            return json.dumps(r, **json_kw)
        # otherwise dump to file
        json.dump(r, output_file, **json_kw)



[docs]    def as_dict(self):
        return self._to_dict()


    def _to_dict(self):
        r = {"ID": self._id, "metadata": {}, "data": []}
        md = r["metadata"]
        # create metadata
        for level in Metadata.LEVELS:
            lvl = []
            for k, v in self.meta.get_level(level).items():
                lvl.append({"ID": k, "values": v})
            md[level] = lvl
        # create data
        r["data"] = {"raw": [], "tabular": []}
        for ds in self.data.datasets:
            if not ds.metadata_ids:
                raise ValueError(f"metadata ids are missing for dataset '{ds.name}'")
            dsx = {"ID": ds.id_, "meta_ID": ds.metadata_ids, "name": ds.name}
            if hasattr(ds, "set_header"):
                # create columns, first row being the header
                # columns = [[f"{n} ({u})"] for n, u in ds.header.columns]
                ncolumns = len(ds.header.columns)
                columns = [[] for i in range(ncolumns)]
                # add data, transposing rows into columns
                for row in ds.observations:
                    for i, column in enumerate(columns):
                        column.append(row[i])
                dsx["table"] = []
                for i in range(ncolumns):
                    values = columns[i]
                    column_item = {
                        "values": values,
                        "name": ds.header.columns[i][0],
                        "units": ds.header.columns[i][1],
                    }
                    if len(values) > 0:
                        column_item["type"] = json_type(values[0])
                    dsx["table"].append(column_item)
                dsx["derived"] = ds.derived
                r["data"]["tabular"].append(dsx)
            else:
                dsx["files"] = ds.filenames
                if ds.derived:
                    dsx["derived"] = ds.derived
                r["data"]["raw"].append(dsx)
        return r

    def from_json(path, require_files=False) -> "Record":
        jdata = json.load(open(path, "r"))
        result = validate(jdata)
        if result is not None:
            raise ValueError(f"While validating [{path}]: {result}")
        rec = Record(id_=jdata["ID"])
        meta = Metadata()
        for level, mdlist in jdata["metadata"].items():
            for md in mdlist:
                meta.add_one(level, md["values"], id_=md["ID"])
        rec.meta = meta
        data = Data()
        # loop over all the files in the "raw" section
        for raw_ds in jdata["data"]["raw"]:
            rds = RawDataset(id_=raw_ds["ID"])
            # set metadata ids from json into Dataset
            rds.set_meta(rec.meta, raw_ds["meta_ID"])
            # add files to dataset, optionally checking for file existence
            for filename in raw_ds["files"]:
                filepath = pathlib.Path(filename)
                if require_files and not filepath.exists():
                    curdir = filepath.cwd()
                    raise ValueError(
                        f"Expected raw data file {filename} " f"not found in {curdir}"
                    )
                rds.add_file(str(filepath))
            # add derived data, if it's present
            if "derived" in raw_ds:
                for field, units, value in raw_ds["derived"]:
                    rds.add_derived(field, units, value)
            # add raw dataset to overall data
            data.add_dataset(rds)
        # loop over all the tables in the "tabular" section
        for tab_ds in jdata["data"]["tabular"]:
            # create new Dataset to hold table
            tds = Dataset(name=tab_ds["name"], id_=tab_ds["ID"])
            # set metadata ids from json into Dataset
            tds.set_meta(rec.meta, tab_ds["meta_ID"])
            # add table to dataset
            table = tab_ds["table"]
            # transpose input column values into rows, with a header
            header = DatasetHeader()
            if len(table) == 0 or len(table[0]["values"]) == 0:
                continue  # nothing in table
            # create all rows
            nrows = len(table[0]["values"])
            rows = [[] for _ in range(nrows)]
            # loop over each column
            for i, column in enumerate(table):
                # add this column to header
                header.add_column(column["name"], column["units"])
                # add values from column to every row
                for value in column["values"]:
                    rows[i].append(value)
            tds.set_header(header)
            for row in rows:
                tds.add_observation(row)
            # add derived data, if it's present
            if "derived" in tab_ds:
                for field, units, value in tab_ds["derived"]:
                    tds.add_derived(field, units, value)
            # add tabular dataset to overall data
            data.add_dataset(tds)
        rec.data = data
        return rec

[docs]class Metadata:
    """Metadata for a Record.

    Use the `add_<section>` methods to add metadata of a given level.
    """

    # use these abbreviations externally
    LVL_LAB, LVL_EXP, LVL_MSR, LVL_SMP = "lab", "experiment", "measurement", "sample"
    # All the levels, in order
    LEVELS = LVL_LAB, LVL_EXP, LVL_MSR, LVL_SMP

    def __init__(self):
        self._meta = {}
        for level in self.LEVELS:
            self._meta[level] = {}


[docs]    def add_one(self, level: str, meta, id_: str = None) -> str:
        """Add one metadata block for a given level.
        """
        if level not in self.LEVELS:
            raise ValueError(f"Level {level} not in {self.LEVELS}")
        uu = str(uuid4()) if id_ is None else id_
        self._meta[level][uu] = meta
        return uu



[docs]    def add(self, group: Dict[str, dict]) -> Dict[str, str]:
        """Add a group of metadata blocks, from different levels.
        This is a convenient way to add all the necessary metadata levels in one call.

        Args:
            group: A dict where key is the name of the metadata level, and
                   value is a dict of metadata fields and values.

        Returns:
            A dict where key is the name of the metadata level, and value is
            the unique identifier (auto-generated) for the provided metadata block
        """
        ids = {}
        for k, v in group.items():
            ids[k] = self.add_one(k, v)
        return ids

[docs]    def set_header(self, header: DatasetHeader):
        self._hdr = header
        n = len(header)
        if self._n is None:
            self._n = n
        else:
            if n != self._n:
                raise ValueError(f"Header length={n} but observation length={self._n}")


    @property
    def header(self):
        return self._hdr


[docs]    def add_observation(self, row):
        """Add a list of values as a single observation.
        """
        self._tbl.append(row)
        n = len(row)
        if self._n is None:
            self._n = n
        else:
            if n != self._n:
                raise ValueError(f"Observation length={n} but header length={self._n}")


    @property
    def observations(self):
        return self._tbl

    def __len__(self):
        return len(self._tbl)
