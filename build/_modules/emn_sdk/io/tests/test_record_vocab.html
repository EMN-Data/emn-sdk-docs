# stdlib
from copy import deepcopy
import json
import logging

# third-party
import pytest

# package
from emn_sdk.io.record_schema import SCHEMA
from emn_sdk.io.model import Metadata
from emn_sdk.io.record_vocab import Vocabulary
from emn_sdk.io import validation
from .util import ckan_info, CKANInfo

# set up logging
_log = logging.getLogger(__name__)


minimal_record = {
    "ID": "72cc11a1-324c-49cd-bf11-083b6acd0571",
    "metadata": {"lab": [], "experiment": [], "measurement": [], "sample": []},
    "data": {"raw": [], "tabular": []},
}

spec1 = {
    Metadata.LVL_MSR: [
        {"if": ("measurement_type", "VIR"), "require": ["Cathode_catalyst_material"]}
    ]
}

spec3 = {
    Metadata.LVL_MSR: [
        {"if": ("measurement_type", "VIR"), "require": ["Cathode_catalyst_material"]}
    ],
    Metadata.LVL_LAB: [
        {"if": ("lab", "LBNL"), "require": ["coffee_roast", "beer_style"]},
        {"if": ("lab", "ANL"), "require": ["pizza_thickness"]},
    ],
}

spec_complex = {
    Metadata.LVL_MSR: [
        {
            "if": ("measurement_type", "VIR"),
            "require": [
                ("Cathode_catalyst_material", {"enum": ["Pt/C", "Au/C", "Hg/C"]})
            ],
        }
    ]
}

        if expect_failure:
            assert failure is not None
        else:
            if failure:
                _log.debug(f"Failure: {failure}")
            assert failure is None

[docs]def test_from_csv(tmp_path):
    # spec1 in CSV form
    table_spec = (
        "date,level,field,value,require,other\n"
        "11/1/19,measurement,measurement_type,VIR,Cathode_catalyst_material,its great\n"
    )
    v = Vocabulary.from_csv(table_spec)
    assert v.specification == spec1

    tempfile = tmp_path / "vocab_from.csv"
    with tempfile.open(mode="w") as f:
        f.write(table_spec)
    v2 = Vocabulary.from_csv(str(tempfile))
    assert v2.specification == spec1




[docs]def test_vocab_from_ckan(ckan_info: CKANInfo):
    c = ckan_info  # alias
    if c.login and c.password and c.dataset_id and c.url and c.apikey:
        v = Vocabulary.from_ckan(c.login, c.password, c.dataset_id, c.url, c.apikey)
        _log.debug(f"Got vocabulary spec from CKAN: {v.specification}")
        # minimal record should always be OK
        failure = validation.validate(minimal_record, v)
        if failure:
            _log.debug(f"Failure: {failure}")
        assert failure is None
    else:
        pytest.skip("Missing some/all CKAN information from environment.")




[docs]def test_empty_table_name():
    table = [
        ["level", "field", "value", "require"],
        [Metadata.LVL_LAB, "-", "x", "author,institution"],
        [Metadata.LVL_EXP, "", "x", "technology_type"],
    ]
    v = Vocabulary.from_table(table)
    rec = {
        "ID": "72cc11a1-324c-49cd-bf11-083b6acd0571",
        "metadata": {
            "lab": [{"ID": "lab-meta-id", "values": {}}],
            "experiment": [{"ID": "exp-meta-id", "values": {}}],
            "measurement": [],
            "sample": [],
        },
        "data": {"raw": [], "tabular": []},
    }
    # no required stuff
    failure = validation.validate(rec, vocab=v)
    assert failure
    # some required stuff
    rec["metadata"]["lab"][0]["values"] = {"author": "Joe", "institution": "Joe's"}
    failure = validation.validate(rec, vocab=v)
    assert failure
    # all the required stuff
    rec["metadata"]["experiment"][0]["values"] = {"technology_type": "LTE"}
    failure = validation.validate(rec, vocab=v)
    assert failure is None
