    def get_env(cls):
        """Fetch values from environment.
        """
        # get vars to fetch from class attributes (dynamic)
        names = [
            k for k in vars(cls).keys() if not k.startswith("_") and not k == "get_env"
        ]
        # transform class attribute into env var, and try to get its value
        for name in names:
            env_name = f"CKAN_{name.upper()}"
            try:
                value = os.environ[env_name]
                setattr(cls, name, value)
            except KeyError:
                _log.warning(
                    f"Could not find '{env_name}' in environment: CKAN "
                    f"tests may not run correctly."
                )
