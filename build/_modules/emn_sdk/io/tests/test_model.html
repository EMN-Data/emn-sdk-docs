[docs]def test_raw_files(tmp_path):
    rec = Record()
    uuids = rec.meta.add(
        {
            rec.meta.LVL_LAB: {"laboratory": "lbnl"},
            rec.meta.LVL_EXP: {"key": "value"},
            rec.meta.LVL_MSR: {"key": "value"},
            rec.meta.LVL_SMP: {"key": "value"},
        }
    )
    temp_files = [tmp_path / f"file{i}.dat" for i in range(3)]
    ds = RawDataset(name="raw1", require_files=False)
    ds.set_meta(rec.meta, uuids)
    # ok, since files are not required to exist
    for p in temp_files:
        ds.add_file(str(p))
    ds = RawDataset(name="raw1")
    # fail since files are required
    ds = RawDataset(name="raw1")  # <-- required
    ds.set_meta(rec.meta, uuids)
    for p in temp_files:
        pytest.raises(IOError, ds.add_file, str(p))
    # create the files
    for p in temp_files:
        with p.open(mode="wb") as f:
            f.write(b"Some bytes")
    # try again, this time should succeed
    ds = RawDataset(name="raw1")  # <-- required
    ds.set_meta(rec.meta, uuids)
    for p in temp_files:
        ds.add_file(str(p))
