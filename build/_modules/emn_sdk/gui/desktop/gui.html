    """

    def __init__(self, url=None, apikey=None, project=None, proxies=None):
        """Initialize the Archive object.

        Args:
            url: URL to connect to CKAN server
            apikey: user inputted API key to access particular projects
            project: Particular project selected for data download or upload
            proxies: optional key for upload/download through proxy servers

        """
        # initial values for params and metadata
        self._url, self._api_key, self._project = url, apikey, project
        self._post_kw = {}
        self.info = {}
        self.uploadurl = ""
        self.downloadpath = ""
        if proxies:
            self._post_kw["proxies"] = proxies
        self._parser_factory = None  # TODO: use this to pick parser

    def _connect(self):
        """

        Connect to CKAN.

        Returns:
            CKAN object configured to the unique API key

        """
        _log.info(f"Connect to CKAN archive at '{self._url}'")
        _log.debug(f"API key for CKAN = '{self._api_key}'")
        return CKAN(self._url, self._api_key)

[docs]    def upload_files(
        self, target, file_or_dir=None, metadata=None, include_source=True
    ):
        """Upload file resources into the target dataset.

        Args:
            target (DatasetResult): Target dataset
            file_or_dir: uploaded file or directory
            metadata (dict): Other key/value pairs added to resource metadata
            include_source (bool): If true, upload pretty much everything in the
                                   input directory

        """
        parser_patterns = patterns.glob_patterns

        if not os.path.isdir(file_or_dir):  # file
            factory = FilenameParserFactory(parser_patterns, glob=True)
            parser = batch_parser.Parser(factory)
            records = parser.parse([file_or_dir])
            basedir = os.path.dirname(file_or_dir)
            other_files = [str(file_or_dir)]
        else:  # directory
            factory = parser_factory.ParserFactory.from_settings(Settings(Settings.find_settings(create=True)))
            # if args.data_type is None:  # auto-detect
            parser_class = factory.detect_parser(file_or_dir)
            if parser_class is None:
                m = QMessageBox()
                m.setText("Unable to parse the data provided into a metadata record.")
                m.setWindowTitle("Upload Error")
                m.setIcon(QMessageBox.Critical)
                m.exec_()
                return
            parser = parser_class()
            records = parser.parse(file_or_dir)
            # parsed_files = records.keys()
            other_files = []
            for root, directories, files in os.walk(str(file_or_dir), topdown=False):
                for name in files:
                    other_files.append(os.path.join(root, name))
            # other_files = [f for f in glob.glob(str(file_or_dir)) if not os.path.isdir(f)]
            basedir = file_or_dir

        if not include_source:
            other_files = None  # do not include source

        ckan = self._connect()
        ckan.set_dataset_info(
            project="api-sandbox", inst="Lawrence Berkeley National Laboratory"
        )
        record_list = []
        # records vs records.values(), if it a string (aka no metadata found), changes to empty records
        if len(records) == 0:
            temp_rec = model.Record()
            d = datetime.date.today()
            t = str(datetime.datetime.now())
            temp_rec.meta.add_one("lab", t, temp_rec._id)
            record_list.append(temp_rec)
        else:
            for rval in records:
                if isinstance(rval, str):
                    record_list.append(model.Record())
                else:
                    record_list.append(rval)
        if _log.isEnabledFor(logging.DEBUG):
            _log.debug(f"uploading to CKAN: records={record_list}")
        ckan.upload(
            target, record_list, other_files, basedir=basedir, use_existing=True
        )



[docs]    def download_dataset(self, overwrite: bool = False, **ckan_kwargs) -> (int, int):
        """Download resources from a dataset from CKAN into a directory.

        Args:
            overwrite: Overwrite flag
            **ckan_kwargs: Passed to `self._ckan.download()`
        Returns:
            (num_files, num_bytes)
        Raises:
            DownloadError
        """
        ckan = self._connect()
        overwrite_action = None if overwrite else "error"
        try:
            result = ckan.download(overwrite=overwrite_action, **ckan_kwargs)
        except KeyError as err:
            raise DownloadError(f"Cannot overwrite file: {err}")
        except CKANError as err:
            raise DownloadError(f"{err}")
        return result

[docs]    def upload_action(self):
        """Upload dataset to CKAN.

        Raises:
            CKANError


        """
        # verify inputs
        if not self._projects.keys().__contains__('None'):
            upload_path = Path(self.uploadDataSource.text())
            if not upload_path.exists():
                self._modal_error(f"data source {upload_path} is not valid")
                return
            ds_name = self.uploadDataTarget.text()
            self._settings.default_path = self.uploadDataSource.text()
            self._settings.save()
            if not ds_name:
                self._modal_error(f"must select a target dataset")
                return
            # perform upload
            t0 = time.time()
            add_to_existing = self.uploadAddFlag.isChecked()
            proj = self.uploadProject.currentText()
            if not add_to_existing:
                try:
                    _log.debug(f"create dataset name='{ds_name}'")
                    self.arch.create_dataset(ds_name)
                except CKANError as err:
                    self._modal_error(f"Creating dataset on CKAN: {err}")
                    return
            _log.debug(f"upload to dataset name='{ds_name}'")
            try:
                self.arch.upload_files(ds_name, upload_path)
            except CKANError as err:
                self._modal_error(f"Upload error from CKAN: {err}")
                return
            dur_s = time.time() - t0
            self._upload_log.append(
                {
                    "path": upload_path,
                    "dataset": ds_name,
                    "project": proj,
                    "append": add_to_existing,
                    "duration": dur_s,
                }
            )
            n = len(self._upload_log)
            upload_message = f"Uploaded {n} file{'s' if n > 1 else ''}"
            if os.path.isdir(upload_path):
                upload_message = f"Uploaded {n} folder{'s' if n > 1 else ''}"
            self.uploadMessages.setText(upload_message)



[docs]    def download_action(self):
        """Download dataset resources from CKAN.

        Raises:
            DownloadError

        """
        if not self._projects.keys().__contains__('None'):
            ds_id = self.downloadDataTarget.text().strip()
            proj = self.downloadProject.currentText()
            path = self.downloadDataDest.text()
            _log.debug(f"Download dataset id={ds_id} path={path}")
            # TODO: optionally make dataset subdir
            if self.downloadNewdirFlag.isChecked():
                if os.path.isdir(path + os.path.sep + ds_id) and not self.downloadOverwriteFlag.isChecked():
                    m = QMessageBox()
                    m.setText("Already a subdirectory with that name in this directory. Please"
                              " check the overwrite box if you would like to overwrite the files.")
                    m.setWindowTitle("Download Error")
                    m.setIcon(QMessageBox.Critical)
                    m.exec_()
                    return
                path = path + os.path.sep + ds_id
                os.mkdir(path)

            t0 = time.time()
            try:
                nfiles, nbytes = self.arch.download_dataset(
                    project=proj, dataset=ds_id, dest_dir=path, overwrite=self.downloadOverwriteFlag.isChecked()
                )
            except DownloadError as err:
                m = QMessageBox()
                m.setText(str(err))
                m.setWindowTitle("Download Error")
                m.setIcon(QMessageBox.Critical)
                m.exec_()
                return
            dur_s = time.time() - t0
            self._download_log.append(
                {
                    "path": str(path),
                    "dataset": ds_id,
                    "project": proj,
                    "duration": dur_s,
                    "nbytes": nbytes,
                    "nfiles": nfiles,
                }
            )
            n = sum(entry["nfiles"] for entry in self._download_log)
            message = f"Downloaded {n} file{'s' if n > 1 else ''}"
            self.downloadMessages.setText(message)

    """

    Main method to run the User Interface program.

    Returns:
        Executed program.

    """
    _log.info(f"Starting GUI")
    # auto-scale the fonts, etc.
    os.environ["QT_AUTO_SCREEN_SCALE_FACTOR"] = "1"
    app = QtWidgets.QApplication(sys.argv)
    try:
        spath = Settings.find_settings()
    except ValueError as err:
        print(f"ERROR: {err}")
        return -1
    if spath is None:
        _log.info("Create new settings file")
        try:
            spath = Settings.create_settings()
            print(spath)
        except ValueError as err:
            print(f"No settings file found, and could not create: {err}")
            return -1
        _log.info(f"Settings file created in '{spath}'")
    main_window = QtWidgets.QMainWindow()
    tabbed_ui = TabbedUI(main_window, settings_path=spath)
    main_window.setCentralWidget(tabbed_ui)
    main_window.show()
    # run app
    result = app.exec_()
    _log.info("Stopping GUI")
